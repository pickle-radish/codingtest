def solution(heights):
    answer = []
    for i in range(len(heights)):
        answer.append(0)
    
    for idx, height in enumerate(heights):
        for i in range(idx, -1, -1):
            if heights[i] > height:
                answer[idx] = i + 1
                break
    return answer
